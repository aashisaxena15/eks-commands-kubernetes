    2  kubectl get svc
    3  kubectl create namespace argocd

    5  kubectl get svc
    6  kubectl get pods
    7  kubectl get pods -n argocd
    8  kubectl get svc -n argocd
    9  kubectl get all -n argocd
   10  kubectl patch svc argocd-server -n argocd -p '{"spec": {"type": "LoadBalancer"}}'

   12  kubectl get all -n argocd
   13  argocd admin initial-password -n argocd
